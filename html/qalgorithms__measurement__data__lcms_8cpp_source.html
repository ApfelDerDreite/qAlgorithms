17 void LCMSData::readCSV(std::string filename,int rowStart, int rowEnd, int colStart, int colEnd, char separator,std::vector<DataType::DataField> variableTypes) {
18 std::ifstream file(filename);
19 std::string line;
20 std::vector<std::vector<std::string>> raw_data;
21
22 // if rowEnd is -1, then set it to the maximum number of rows
23 if (rowEnd == -1) {
24 rowEnd = 1000000;
25 }
26
27 // if colEnd is -1, then set it to the maximum number of columns
28 if (colEnd == -1) {
29 colEnd = 1000000;
30 }
31
32 // find the DataFiled::SCANNUMBER as this is the key for the LCMSData map
33 int scanNumberIndex = -1;
34 for (int i = 0; i < variableTypes.size(); i++) {
35 if (variableTypes[i] == DataType::DataField::SCANNUMBER) {
36 scanNumberIndex = i;
37 break;
38 }
39 }
40
41 // find the DataFiled::RETENTIONTIME
42 int retentionTimeIndex = -1;
43 for (int i = 0; i < variableTypes.size(); i++) {
44 if (variableTypes[i] == DataType::DataField::RETENTIONTIME) {
45 retentionTimeIndex = i;
46 break;
47 }
48 }
49
50 // find the DataFiled::MZ
51 int mzIndex = -1;
52 for (int i = 0; i < variableTypes.size(); i++) {
53 if (variableTypes[i] == DataType::DataField::MZ) {
54 mzIndex = i;
55 break;
56 }
57 }
58
59 // find the DataFiled::INTENSITY
60 int intensityIndex = -1;
61 for (int i = 0; i < variableTypes.size(); i++) {
62 if (variableTypes[i] == DataType::DataField::INTENSITY) {
63 intensityIndex = i;
64 break;
65 }
66 }
67
68 // read the file
69 int rowCounter = 0;
70 int colCounter = 0;
71 while (std::getline(file, line) && rowCounter < rowEnd) {
72 if (rowCounter >= rowStart) {
73 std::vector<std::string> row;
74 std::stringstream lineStream(line);
75 std::string cell;
76 colCounter = 0;
77 while (std::getline(lineStream, cell, separator)) {
78 if (colCounter >= colStart && colCounter < colEnd) {
79 row.push_back(cell);
80 }
81 colCounter++;
82 }
83 raw_data.push_back(row);
84 }
85 rowCounter++;
86 }
87 file.close();
88
89 // check if the raw data has same number of columns as the variableTypes
90 if (raw_data[0].size() != variableTypes.size()) {
91 std::cerr << "The number of columns in the raw data does not match the number of variable types" << std::endl;
92 return;
93 }
94
95 // transfere the raw data to the data map
96 for (int i = 0; i < raw_data.size(); i++) {
97 // get/set the scan number
98 int scanNumber = std::stoi(raw_data[i][scanNumberIndex]);
99 int subID = 0;
100 this->data[scanNumber][subID].scanNumber = scanNumber;
101
102 // get/set the retention time
103 this->data[scanNumber][subID].retentionTime = std::stod(raw_data[i][retentionTimeIndex]);
104
105 // get/set the mz
106 this->data[scanNumber][subID].mz.push_back(std::stod(raw_data[i][mzIndex]));
107
108 // get/set the intensity
109 this->data[scanNumber][subID].intensity.push_back(std::stod(raw_data[i][intensityIndex]));
110
111 // get/set optional variable types
112 // following fields are allowed:
113 // DataField::IONIZATIONMODE,
114 // DataField::MEASUREMENTMODE,
115 // DataField::MSLEVEL
116 for (int j = 0; j < variableTypes.size(); j++) {
117 if (variableTypes[j] == DataType::DataField::IONIZATIONMODE) {
118 this->data[scanNumber][subID].ionizationMode = (raw_data[i][j] == "Positive") ? DataType::IonizationMode::Positive : DataType::IonizationMode::Negative;
119 } else if (variableTypes[j] == DataType::DataField::MEASUREMENTMODE) {
120 this->data[scanNumber][subID].measurementMode = (raw_data[i][j] == "Profile") ? DataType::MeasurementMode::Profile : DataType::MeasurementMode::Centroid;
121 } else if (variableTypes[j] == DataType::DataField::MSLEVEL) {
122 this->data[scanNumber][subID].msLevel = (raw_data[i][j] == "MS1") ? DataType::MSLevel::MS1 : DataType::MSLevel::MS2;
123 }
124 }
125 }

138 void LCMSData::cutData() {
139 // create a new data map to store the updated data
140 std::unordered_map<int, std::unordered_map<int, DataType::LC_MS>> updatedData;
141 // iterate over all data sets
142 for (auto it = this->data.begin(); it != this->data.end(); it++) {
143 // iterate over all sub data sets
144 for (auto it2 = it->second.begin(); it2 != it->second.end(); it2++) {
145 // read the indices where the data needs to be cut
146 std::vector<size_t> indices = this->MeasurementData::cutData(it2->second.mz, it2->second.intensity);
147
148 if (indices.size() == 0) {
149 // no cut is needed. however, zerofilling will add a separator at least at the beginning of the data.
150 updatedData[it->first][it2->first] = it2->second;
151 } else {
152 // cut the data
153 int subID = 0;
154 // add the last index as the size of the data
155 indices.push_back(it2->second.mz.size());
156 for (int i = 0; i < indices.size() - 1; i++) {
157 // create a new data subset
158 DataType::LC_MS newSubset = it2 -> second;
159 // adjust the mz and intensity vectors
160 newSubset.mz = std::vector<double>(it2->second.mz.begin() + indices[i] + 1, it2->second.mz.begin() + indices[i+1]);
161 newSubset.intensity = std::vector<double>(it2->second.intensity.begin() + indices[i] + 1, it2->second.intensity.begin() + indices[i+1]);
162 updatedData[it->first][subID] = newSubset;
163 subID++;
164 }
165 }
166 }
167 }
168
169 // update the data map
170 this->data = updatedData;
171 }
