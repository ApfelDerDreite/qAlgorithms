73 bool Peak::validate_coefficients(
74 const double df,
75 const double b1,
76 const double b2,
77 const double b3,
78 const double xIndex,
79 double& apex_position,
80 double& valley_position) const {
81 /*
82 This functions checks if a set of regression coefficients (b0, b1, b2, b3) are valid
83 and describe an analytical peak after exp-transform.
84 # valid peak case 1: b2<0 & b3<0 & apex in window
85 # valid peak case 2a: b2>0 & b3<0 & b1>0 & |valley - apex| > 2 & apex in window
86 # valud peak case 2b: b2<0 & b3>0 & b1<0 & |valley - apex| > 2 & apex in window
87 */
88
89 /* creating unique checksum */
90 int key = (b1 < 0 ? 4 : 0) + (b2 < 0 ? 2 : 0) + (b3 < 0 ? 1 : 0);
91
92 switch (key) {
93 case 7: // Case 1a: apex left
94 return calcApex_position(df, b1, b2, xIndex, apex_position);
95 case 3: // Case 1b: apex right
96 return calcApex_position(df, b1, b3, xIndex, apex_position);
97 case 6: // Case 2a: apex left | valley right
98 return calcApexValley_position(df, b1, b2, b3, xIndex, apex_position, valley_position);
99 case 1: // Case 2b: apex right | valley left
100 return calcApexValley_position(df, b1, b3, b2, xIndex, apex_position, valley_position);
101 default:
102 return false;
103 };
104 }
